from picamera import PiCamera
from time import sleep

camera = PiCamera()
frame = 1

def takePhoto(): 
    camera.start_preview()
    sleep(2)
    camera.capture('/home/pi/animation/frame%03d.jpg' % frame)
    global frame += 1
    camera.stop_preview()

elif event.code == aBtn:
    takePhoto()
