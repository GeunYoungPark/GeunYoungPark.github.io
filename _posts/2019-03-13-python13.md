```python
from picamera import PiCamera
from time import sleep

camera = PiCamera()
frame = 1

def takePhoto(): 
    global frame
    camera.start_preview()
    sleep(2)
    camera.capture('/home/pi/kyanime/frame%03d.jpg' % frame)
    frame += 1
    camera.stop_preview()
```
